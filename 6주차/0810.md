## swea 1954.달팽이 숫자

```python
for i in range(10):                                                             #테스트케이스 개수만큼 for문 실행
    thisNum = int(input())                                                      #현재 테스트케이스 순서
    thisCaseList=[list(map(int,input().split())) for _ in range(100)]           #테스트케이스 100x100개에서 한 줄을 한 리스트로 만들어 2차원 배열로 만듦
    maxSum=0                                                                    #가장 큰 합계를 저장하기 위한 변수 생성
    rightDiagonal=0                                                             #대각선 두 개의 합계를 저장하기 위한 변수 생성
    leftDiagonal=0
    for k in range(100):                                                        #k를 열, q를 행으로 지정하고
        rowSum=0                                                                #해당 열의 합계, 해당 행의 합계를 저장하기 위한 변수 생성
        columnSum=0
        for q in range(100):                                                    #각각의 좌표에 따라 rowSum과 columnSum에 값을 추가하고
            rowSum+=thisCaseList[k][q]
            columnSum+=thisCaseList[q][k]
            if k==q:                                                            #대각선에 해당하는 좌표일 경우 각각의 변수에 값을 추가함
                rightDiagonal+=thisCaseList[k][q]
            elif k+q==9:
                leftDiagonal+=thisCaseList[k][q]
        if rowSum>columnSum and rowSum>maxSum:                                  #q가 속해 있는 for문이 끝나면 해당 q행 k열의 모든 좌표에 있는 값이 더해진 이후므로
            maxSum=rowSum                                                       #rowSum과 columnSum, rightDiagonal과 leftDiagonal, 마지막으로
        elif columnSum>rowSum and columnSum>maxSum:                             #이전의 maxNum과 비교하여 가장 높은 값을 maxSum에 넣음
            maxSum=columnSum
    if rightDiagonal>maxSum and rightDiagonal>leftDiagonal:
        maxSum=rightDiagonal
    elif leftDiagonal>maxSum and leftDiagonal>rightDiagonal:
        maxSum=leftDiagonal
    print(maxSum)                                                               #maxSum 출력
```
