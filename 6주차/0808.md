```python
#10개의 케이스를 입력받음
for i in range(10):
    #케이스의 길이를 입력받음
    caseLen=int(input())
    #테스트케이스를 배열로 입력받음
    caseList=list(map(int,input().split()))
    #조망권이 확보되는 지역을 추가할 변수 생성
    goodView=0
    #좌우로 2칸씩 모두 비어있어야 조망권이 확보되는데, 빌딩은 중간에 뚫려있지 않으므로
    #해당 층보다 좌우로 모두 낮아야 조망권이 확보됨
    for k in range(2,caseLen-2):
        #좌우로 2칸씩에 해당하는 테스트케이스를 배열로 생성함
        aroundList=[caseList[k-2],caseList[k-1],caseList[k+1],caseList[k+2]]
        #버블 정렬 이용
        #i번째 항목이 i+1번째 항목보다 크다면 i+1번째로 옮겨가게 됨
        #맨 마지막 항목이 가장 큰 수를 가진 항목이 됨
        for a in range(3):
            if aroundList[a]>aroundList[a+1]:
                aroundList[a],aroundList[a+1]=aroundList[a+1],aroundList[a]
        #마지막 항목이 해당 테스트케이스보다 높다면, 조망권이 확보되는 층은 하나도 없음
        if aroundList[3]>=caseList[k]:
            pass
        #마지막 항목이 해당 테스트케이스보다 낮다면, 둘의 차이만큼 조망권이 확보되는 층이 있음
        #조망권이 확보되는 층의 개수를 goodView에 추가함
        else:
            goodView+=caseList[k]-aroundList[3]
    #모든 조망권 층을 더한 goodView 를 출력함
    print(f'#{i+1} {goodView}')
```
